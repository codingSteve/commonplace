
\newcommand{\hugeQuote}[6]{
	\refstepcounter{section}
	\addcontentsline{toc}{section}{\protect\numberline{\thesection}#2}
	\label{#1}
	
	\begin{tabularx}{\textwidth}{lr}
  		\multicolumn{2}{p{1.0\textwidth} }{\huge{#2}\huge{#3}}
  		\tabularnewline \tabularnewline
  		\ifthenelse{\equal{}{#5}}{}{cf. \ref{#5}}& #4 \index{#4}
  		\ifthenelse{\equal{}{#6}}{}{\tabularnewline \tabularnewline \multicolumn{2}{p{1\textwidth}}{#6}
  		}
	\end{tabularx}
	\bigskip
	\hrulefill
	\vfill
}

\newcommand{\HugeQuote}[6]{
	\refstepcounter{section}
	\addcontentsline{toc}{section}{\protect\numberline{\thesection}#2}
	\vspace*{\fill}
	\label{#1}
	
	\begin{tabularx}{\textwidth}{ lr}
  		\multicolumn{2}{p{1.0\textwidth} }{\huge{#2}\huge{#3}}
  		\tabularnewline \tabularnewline
  		\ifthenelse{\equal{}{#5}}{}{cf. \ref{#5}}& #4 \index{#4}
  		\tabularnewline
  		\tabularnewline
  		\multicolumn{2}{p{1\textwidth}}{#6} 
	\end{tabularx}
	\bigskip
	\hrulefill
	\vspace*{\fill}
	\newpage
}
