
\newcommand{\standardQuote}[5]{
	\refstepcounter{section}
	\addcontentsline{toc}{section}{\protect\numberline{\thesection}}
	
	%\marginnote{\large{\thesection}}
	\begin{tabularx}{\textwidth}{ lr}
  		\multicolumn{2}{p{1.0\textwidth}}{\label{#1}#2}
  		\tabularnewline \tabularnewline
  		\ifthenelse{\equal{}{#4}}{}{cf. \ref{#4}}& #3 \index{#3}
  		\tabularnewline
  		\tabularnewline
  		\multicolumn{2}{p{1\textwidth}}{#5} 
  	\end{tabularx}
	\bigskip
	\hrulefill
}

\newcommand{\hugeQuote}[6]{
	\standardQuote{#1}{\huge{\thesection}: \huge{#2}\huge{#3}}{#4}{#5}{#6}
	\vfill
}
