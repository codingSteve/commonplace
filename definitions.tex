
\newcommand{\standardQuote}[5]{
	\refstepcounter{section}
	\addcontentsline{toc}{section}{\protect\numberline{\thesection}}
	
	\begin{tabularx}{\linewidth}{ lr}
  		\multicolumn{2}{p{1.0\linewidth}}{\label{#1}#2}
  		\tabularnewline \tabularnewline
  		\ifthenelse{\equal{}{#4}}{}{\S \ref{#4}}& #3 (\thesection) \index{#3}
  		\tabularnewline
  		\tabularnewline
  		\multicolumn{2}{p{1\linewidth}}{#5} 
  	\end{tabularx}
	% \bigskip
	
	\hrulefill
}

\newcommand{\hugeQuote}[6]{
	\standardQuote{#1}{\huge{#2}\huge{#3}}{#4}{#5}{#6}
	\vfill
}

}
